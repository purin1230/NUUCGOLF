const SHEET_NAME = '報名名單'; const EMAIL_TO = 'emma12301230@gmail.com'; function doPost(e) { const ss = SpreadsheetApp.getActiveSpreadsheet(); let sheet = ss.getSheetByName(SHEET_NAME); if (!sheet) { sheet = ss.insertSheet(SHEET_NAME); } if (sheet.getLastRow() === 0) { sheet.appendRow(['報名時間', '姓名', '電話', 'Email', '企家班屆數', '性別', 'Line ID', '球齡/程度', '是否需要安排同組', '備註']); } const p = e.parameter || {}; const row = [ new Date(), p.name || '', p.phone || '', p.email || '', p.classYear || '', p.gender || '', p.lineId || '', p.level || '', p.groupNeed || '', p.note || '' ]; sheet.appendRow(row); const subject = '🏌️ 第28屆政大企家班師徒隊｜新報名通知'; const body = '有新的報名資料：\n\n' + '姓名：' + (p.name || '') + '\n' + '電話：' + (p.phone || '') + '\n' + 'Email：' + (p.email || '') + '\n' + '企家班屆數：' + (p.classYear || '') + '\n' + '性別：' + (p.gender || '') + '\n' + 'Line ID：' + (p.lineId || '') + '\n' + '球齡/程度：' + (p.level || '') + '\n' + '是否需要安排同組：' + (p.groupNeed || '') + '\n' + '備註：' + (p.note || '') + '\n\n' + '報名時間：' + new Date(); MailApp.sendEmail(EMAIL_TO, subject, body); return ContentService .createTextOutput(JSON.stringify({ result: 'success' })) .setMimeType(ContentService.MimeType.JSON); }
